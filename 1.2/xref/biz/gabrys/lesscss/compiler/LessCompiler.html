13  package biz.gabrys.lesscss.compiler;
14  
15  import java.io.File;
16  
17  /**
18   * Responsible for compiling <a href="http://lesscss.org/">Less</a> sources to the
19   * <a href="http://www.w3.org/Style/CSS/">CSS</a> code.
20   * @since 1.0
21   */
22  public interface LessCompiler {
23  
24      /**
25       * Compiles a <a href="http://lesscss.org/">Less</a> source file to the
26       * <a href="http://www.w3.org/Style/CSS/">CSS</a> code with default options.
27       * @param source the source file.
28       * @return the <a href="http://www.w3.org/Style/CSS/">CSS</a> code.
29       * @throws CompilerException if an error occurred during compilation process.
30       * @since 1.0
31       * @see CompilerOptions
32       */
33      String compile(File source) throws CompilerException;
34  
35      /**
36       * Compiles a <a href="http://lesscss.org/">Less</a> source file to the
37       * <a href="http://www.w3.org/Style/CSS/">CSS</a> code.
38       * @param source the source file.
39       * @param options the compiler options.
40       * @return the <a href="http://www.w3.org/Style/CSS/">CSS</a> code.
41       * @throws CompilerException if an error occurred during compilation process.
42       * @since 1.0
43       */
44      String compile(File source, CompilerOptions options) throws CompilerException;
45  }
