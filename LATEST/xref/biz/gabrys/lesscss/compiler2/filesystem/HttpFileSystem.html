32  /**
33   * <p>
34   * Represents a file system accessible via <a href="https://www.w3.org/Protocols/">HTTP</a> and HTTPS protocols. The
35   * file system handles HTTP codes:
36   * </p>
37   * <ul>
38   * <li>200 OK</li>
39   * <li>301 Moved Permanently</li>
40   * <li>302 Temporary Redirect</li>
41   * <li>303 See Other</li>
42   * <li>404 Not Found</li>
43   * </ul>
44   * <p>
45   * Example paths:
46   * </p>
47   * <ul>
48   * <li>http://example.org/style.less</li>;
49   * <li>https://example.org/styles/main.less</li>
50   * </ul>
51   * @since 2.0.0
52   */
53  public class HttpFileSystem implements FileSystem {
54  
55      /**
56       * Stores HTTP {@code OK} (200) and {@code NOT FOUND} (404) codes.
57       * @since 2.0.0
58       * @see HttpURLConnection#HTTP_OK
59       * @see HttpURLConnection#HTTP_NOT_FOUND
60       */
61      protected static final Collection<Integer> OK_NOTFOUND_CODES = Collections
62              .unmodifiableCollection(Arrays.asList(HttpURLConnection.HTTP_OK, HttpURLConnection.HTTP_NOT_FOUND));
63  
64      /**
65       * Stores HTTP {@code MOVED PERMANENTLY} (301), {@code MOVED TEMPORARILY} (302), and {@code SEE OTHER} (303) codes.
66       * @since 2.0.0
67       * @see HttpURLConnection#HTTP_MOVED_PERM
68       * @see HttpURLConnection#HTTP_MOVED_TEMP
69       * @see HttpURLConnection#HTTP_SEE_OTHER
70       */
71      protected static final Collection<Integer> REDIRECT_CODES = Collections.unmodifiableCollection(
72              Arrays.asList(HttpURLConnection.HTTP_MOVED_PERM, HttpURLConnection.HTTP_MOVED_TEMP, HttpURLConnection.HTTP_SEE_OTHER));
73  
74      /**
75       * Stores HTTP {@code OK} (200), {@code NOT FOUND} (404), {@code MOVED PERMANENTLY} (301), {@code MOVED TEMPORARILY}
76       * (302), and {@code SEE OTHER} (303) codes.
77       * @since 2.0.0
78       * @see HttpURLConnection#HTTP_OK
79       * @see HttpURLConnection#HTTP_NOT_FOUND
80       * @see HttpURLConnection#HTTP_MOVED_PERM
81       * @see HttpURLConnection#HTTP_MOVED_TEMP
82       * @see HttpURLConnection#HTTP_SEE_OTHER
83       */
84      protected static final Collection<Integer> OK_NOTFOUND_REDIRECT_CODES;
85  
86      static {
87          final Collection<Integer> codes = new ArrayList<Integer>(5);
88          codes.addAll(OK_NOTFOUND_CODES);
89          codes.addAll(REDIRECT_CODES);
90          OK_NOTFOUND_REDIRECT_CODES = Collections.unmodifiableCollection(codes);
91      }
92  
93      private static final Pattern CHARSET_PATTERN = Pattern.compile("(?i)\\bcharset=\\s*\"?([^\\s;\"]*)");
94  
95      /**
96       * Constructs a new instance.
97       * @since 2.0.0
98       */
99      public HttpFileSystem() {
100         // do nothing
101     }
102 
103     @Override
104     public void configure(final Map<String, String> parameters) {
105         // do nothing
106     }
107 
108     @Override
109     public boolean isSupported(final String path) {
110         return path.startsWith("http://") || path.startsWith("https://");
111     }
112 
113     @Override
114     public String normalize(final String path) throws URISyntaxException {
115         return new URI(path).normalize().toString();
116     }
117 
118     @Override
119     public String expandRedirection(final String path) throws IOException, URISyntaxException {
120         HttpURLConnection connection = null;
121         String redirectedPath;
122         try {
123             connection = makeConnection(new URL(path), false);
124             final int responseCode = connection.getResponseCode();
125             validateResponseCode(responseCode, OK_NOTFOUND_REDIRECT_CODES);
126             if (OK_NOTFOUND_CODES.contains(responseCode)) {
127                 return path;
128             }
129             final String location = connection.getHeaderField("Location");
130             redirectedPath = connection.getURL().toURI().resolve(location).toString();
131         } finally {
132             disconnect(connection);
133         }
134         return expandRedirection(redirectedPath);
135     }
136 
137     @Override
138     public boolean exists(final String path) throws IOException {
139         HttpURLConnection connection = null;
140         try {
141             connection = makeConnection(new URL(path), false);
142             final int responseCode = connection.getResponseCode();
143             validateResponseCode(responseCode, OK_NOTFOUND_CODES);
144             return responseCode == HttpURLConnection.HTTP_OK;
145         } finally {
146             disconnect(connection);
147         }
148     }
149 
150     @Override
151     public FileData fetch(final String path) throws IOException {
152         HttpURLConnection connection = null;
153         try {
154             connection = makeConnection(new URL(path), true);
155             validateResponseCode(connection.getResponseCode(), Arrays.asList(HttpURLConnection.HTTP_OK));
156             final String responseEncoding = readEncoding(connection);
157             try {
158                 return new FileData(IOUtils.toByteArray(connection.getInputStream()), responseEncoding);
159             } catch (final IOException e) {
160                 throw new IOException("cannot download file", e);
161             }
162         } finally {
163             disconnect(connection);
164         }
165     }
166 
167     /**
168      * Opens and configures a new {@link HttpURLConnection} instance that represents a connection to the remote object
169      * referred to by the URL.
170      * @param url the URL (cannot be {@code null}).
171      * @param fetchResponseBody whether the connection should fetch response body ({@code GET} request) or not
172      *            ({@code HEAD} request).
173      * @return the new configured {@link HttpURLConnection} instance.
174      * @throws IOException if an I/O exception occurs.
175      * @since 2.0.0
176      */
177     protected HttpURLConnection makeConnection(final URL url, final boolean fetchResponseBody) throws IOException {
178         final HttpURLConnection connection = openConnection(url);
179         connection.setInstanceFollowRedirects(false);
180         connection.setRequestMethod(fetchResponseBody ? "GET" : "HEAD");
181         return connection;
182     }
183 
184     /**
185      * Creates a new {@link HttpURLConnection} instance that represents a connection to the remote object referred to by
186      * the URL. The reason why this method has been extracted is allow stubbing and mocking connection objects in tests.
187      * @param url the URL (cannot be {@code null}).
188      * @return the new {@link HttpURLConnection} instance.
189      * @throws IOException if an I/O exception occurs.
190      * @since 2.0.0
191      */
192     protected HttpURLConnection openConnection(final URL url) throws IOException {
193         return (HttpURLConnection) url.openConnection();
194     }
195 
196     /**
197      * Validates if a response code is in an allowed set.
198      * @param responseCode the tested response code.
199      * @param validCodes the collection with allowed response codes (cannot be {@code null}).
200      * @throws IOException if the response code is not in the allowed set.
201      * @since 2.0.0
202      */
203     protected void validateResponseCode(final int responseCode, final Collection<Integer> validCodes) throws IOException {
204         if (!validCodes.contains(responseCode)) {
205             final List<Integer> supportedCodes = new ArrayList<Integer>(validCodes);
206             Collections.sort(supportedCodes);
207             throw new IOException(String.format("response HTTP status code %s is not allowed (supports only: %s)", responseCode,
208                     supportedCodes.toString().replaceAll("\\[|\\]", "")));
209         }
210     }
211 
212     /**
213      * Returns a connection encoding or {@code null} if the {@code content-type} header has incorrect format.
214      * @param connection the connection (cannot be {@code null}).
215      * @return the connection encoding or {@code null} if the {@code content-type} header has incorrect format.
216      * @since 2.0.0
217      */
218     protected String readEncoding(final HttpURLConnection connection) {
219         final String contentType = connection.getContentType();
220         if (contentType != null) {
221             final Matcher matcher = CHARSET_PATTERN.matcher(contentType);
222             if (matcher.find()) {
223                 return matcher.group(1).trim().toUpperCase(Locale.ROOT);
224             }
225         }
226         return null;
227     }
228 
229     /**
230      * Disconnects a connection unconditionally.
231      * @param connection the connection to close (can be {@code null} or already closed).
232      * @since 2.0.0
233      */
234     protected void disconnect(final HttpURLConnection connection) {
235         if (connection == null) {
236             return;
237         }
238         try {
239             connection.disconnect();
240         } catch (final Exception e) {
241             // do nothing
242         }
243     }
