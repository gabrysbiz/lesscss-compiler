13  package biz.gabrys.lesscss.compiler2.util.io;
14  
15  import java.io.File;
16  import java.io.IOException;
17  
18  /**
19   * Responsible for creating temporary files in a system default temporary directory.
20   * @since 2.0.0
21   */
22  public class SystemDefaultTemporaryFileFactory implements TemporaryFileFactory {
23  
24      /**
25       * The prefix string used in generating the temporary files names.
26       * @since 2.0.0
27       */
28      public static final String FILENAME_PREFIX = "gabrys-lesscss-compiler-";
29      /**
30       * The suffix string used in generating the temporary files names.
31       * @since 2.0.0
32       */
33      public static final String FILENAME_SUFFIX = ".less";
34  
35      @Override
36      public File create() throws IOException {
37          return File.createTempFile(FILENAME_PREFIX, FILENAME_SUFFIX);
38      }
39  }
