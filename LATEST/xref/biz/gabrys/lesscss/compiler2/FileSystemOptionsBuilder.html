13  package biz.gabrys.lesscss.compiler2;

15  import java.util.ArrayList;
16  import java.util.LinkedHashSet;
17  import java.util.List;
18  import java.util.Map;
19  import java.util.Set;
20  
21  import biz.gabrys.lesscss.compiler2.filesystem.ClassPathFileSystem;
22  import biz.gabrys.lesscss.compiler2.filesystem.FileSystem;
23  import biz.gabrys.lesscss.compiler2.filesystem.FtpFileSystem;
24  import biz.gabrys.lesscss.compiler2.filesystem.HttpFileSystem;
25  import biz.gabrys.lesscss.compiler2.filesystem.LocalFileSystem;

27  /**
28   * <p>
29   * Responsible for creating value of the {@link LessOptions#getFileSystems() file system} options.
30   * </p>
31   * <p>
32   * Example code:
33   * </p>
34   * 
35   * <pre>
36   * List&lt;{@link FileSystemOption}&gt; fileSystems;
37   * // create a list with standard file systems
38   * fileSystems = new {@link #FileSystemOptionsBuilder() FileSystemOptionsBuilder}().{@link #appendHttp() appendHttp}().{@link #appendLocal() appendLocal}().{@link #build() build}();
39   * 
40   * // create a list with standard file systems (short version)
41   * fileSystems = new {@link #FileSystemOptionsBuilder() FileSystemOptionsBuilder}().{@link #appendStandard() appendStandard}().{@link #build() build}();
42   * 
43   * // create a list with  file system
44   * class CustomFileSystem implements {@link FileSystem} {
45   *      // omitted implementation
46   * }
47   * 
48   * fileSystems = new {@link #FileSystemOptionsBuilder() FileSystemOptionsBuilder}().{@link #appendCustom(Class) appendCustom}(CustomFileSystem.class).{@link #build() build}();
49   * </pre>
50   * <p>
51   * <strong>Warning</strong>: the {@link #appendLocal() local file system} should be appended as last or not at all.
52   * </p>
53   * @since 2.0.0
54   * @see FileSystemOptionBuilder
55   * @see LessOptions
56   */
57  public class FileSystemOptionsBuilder {
58  
59      private final Set<FileSystemOption> options = new LinkedHashSet<>();
60  
61      /**
62       * Constructs a new instance.
63       * @since 2.0.0
64       */
65      public FileSystemOptionsBuilder() {
66          // do nothing
67      }
68  
69      /**
70       * Appends standard file systems with no parameters at the end. If any file system (with no parameters) was appended
71       * before, then it will be removed and appended again. Appended file systems:
72       * <ol>
73       * <li>{@link HttpFileSystem}</li>
74       * <li>{@link FtpFileSystem} (only if <a href="https://commons.apache.org/proper/commons-net/">Apache Commons
75       * Net</a> library in the class path)</li>
76       * <li>{@link LocalFileSystem}</li>
77       * </ol>
78       * @return {@code this} builder.
79       * @since 2.0.0
80       */
81      public FileSystemOptionsBuilder appendStandard() {
82          if (FtpFileSystem.isAvailable()) {
83              return appendHttp().appendFtp().appendLocal();
84          }
85          return appendHttp().appendLocal();
86      }
87  
88      /**
89       * Appends the {@link LocalFileSystem} with no parameters at the end. If the file system (with no parameters) was
90       * appended before, then it will be removed and appended again.
91       * @return {@code this} builder.
92       * @since 2.0.0
93       */
94      public FileSystemOptionsBuilder appendLocal() {
95          appendFileSystem(new FileSystemOption(LocalFileSystem.class));
96          return this;
97      }
98  
99      /**
100      * Appends the {@link HttpFileSystem} with no parameters at the end. If the file system (with no parameters) was
101      * appended before, then it will be removed and appended again.
102      * @return {@code this} builder.
103      * @since 2.0.0
104      */
105     public FileSystemOptionsBuilder appendHttp() {
106         appendFileSystem(new FileSystemOption(HttpFileSystem.class));
107         return this;
108     }
109 
110     /**
111      * <p>
112      * Appends the {@link FtpFileSystem} with no parameters at the end. If the file system (with no parameters) was
113      * appended before, then it will be removed and appended again.
114      * </p>
115      * <p>
116      * <strong>Warning</strong>: requires <a href="https://commons.apache.org/proper/commons-net/">Apache Commons
117      * Net</a> library in the class path.
118      * </p>
119      * @return {@code this} builder.
120      * @since 2.0.0
121      */
122     public FileSystemOptionsBuilder appendFtp() {
123         appendFileSystem(new FileSystemOption(FtpFileSystem.class));
124         return this;
125     }
126 
127     /**
128      * Appends the {@link ClassPathFileSystem} with no parameters at the end. If the file system (with no parameters)
129      * was appended before, then it will be removed and appended again.
130      * @return {@code this} builder.
131      * @since 2.0.0
132      */
133     public FileSystemOptionsBuilder appendClassPath() {
134         appendFileSystem(new FileSystemOption(ClassPathFileSystem.class));
135         return this;
136     }
137 
138     /**
139      * Appends file systems specified by classes with no parameters at the end. If any file system (with no parameters)
140      * was appended before, then it will be removed and appended again.
141      * @param classes the file system classes.
142      * @return {@code this} builder.
143      * @throws IllegalArgumentException is the file system classes are equal to {@code null}.
144      * @since 2.0.0
145      */
146     public FileSystemOptionsBuilder appendCustom(@SuppressWarnings("unchecked") final Class<? extends FileSystem>... classes) {
147         if (classes == null) {
148             throw new IllegalArgumentException("File system classes cannot be null");
149         }
150         for (final Class<? extends FileSystem> clazz : classes) {
151             if (clazz != null) {
152                 appendFileSystem(new FileSystemOption(clazz));
153             }
154         }
155         return this;
156     }
157 
158     /**
159      * Appends a file system specified by class with no parameters at the end. If the file system (with no parameters)
160      * was appended before, then it will be removed and appended again.
161      * @param clazz the file system class.
162      * @return {@code this} builder.
163      * @throws IllegalArgumentException is the file system class is equal to {@code null}.
164      * @since 2.0.0
165      */
166     public FileSystemOptionsBuilder appendCustom(final Class<? extends FileSystem> clazz) {
167         if (clazz == null) {
168             throw new IllegalArgumentException("File system class cannot be null");
169         }
170         appendFileSystem(new FileSystemOption(clazz));
171         return this;
172     }
173 
174     /**
175      * Appends a file system specified by class with parameters at the end. If the file system with the same parameters
176      * was appended before, then it will be removed and appended again.
177      * @param clazz the file system class.
178      * @param parameters the configuration parameters.
179      * @return {@code this} builder.
180      * @throws IllegalArgumentException is the file system class or parameters are equal to {@code null}.
181      * @since 2.0.0
182      */
183     public FileSystemOptionsBuilder appendCustom(final Class<? extends FileSystem> clazz, final Map<String, String> parameters) {
184         if (clazz == null) {
185             throw new IllegalArgumentException("File system class cannot be null");
186         }
187         if (parameters == null) {
188             throw new IllegalArgumentException("Parameters cannot be null");
189         }
190         appendFileSystem(new FileSystemOption(clazz, parameters));
191         return this;
192     }
193 
194     /**
195      * Appends file system options at the end. If any file system option was appended before, then it will be removed
196      * and appended again.
197      * @param options the file system options.
198      * @return {@code this} builder.
199      * @throws IllegalArgumentException is the options is equal to {@code null}.
200      * @since 2.0.0
201      */
202     public FileSystemOptionsBuilder appendCustom(final FileSystemOption... options) {
203         if (options == null) {
204             throw new IllegalArgumentException("File system options cannot be null");
205         }
206         for (final FileSystemOption option : options) {
207             if (option != null) {
208                 appendFileSystem(option);
209             }
210         }
211         return this;
212     }
213 
214     /**
215      * Appends file system options at the end. If any file system option was appended before, then it will be removed
216      * and appended again.
217      * @param options the file system options.
218      * @return {@code this} builder.
219      * @throws IllegalArgumentException is the options is equal to {@code null}.
220      * @since 2.0.0
221      */
222     public FileSystemOptionsBuilder appendCustom(final Iterable<? extends FileSystemOption> options) {
223         if (options == null) {
224             throw new IllegalArgumentException("File system options cannot be null");
225         }
226         for (final FileSystemOption option : options) {
227             if (option != null) {
228                 appendFileSystem(option);
229             }
230         }
231         return this;
232     }
233 
234     /**
235      * Appends a file system option at the end. If the file system option was appended before, then it will be removed
236      * and appended again.
237      * @param option the file system option.
238      * @return {@code this} builder.
239      * @throws IllegalArgumentException is the fileSystemOption is equal to {@code null}.
240      * @since 2.0.0
241      */
242     public FileSystemOptionsBuilder appendCustom(final FileSystemOption option) {
243         if (option == null) {
244             throw new IllegalArgumentException("File system option cannot be null");
245         }
246         appendFileSystem(option);
247         return this;
248     }
249 
250     private void appendFileSystem(final FileSystemOption option) {
251         options.remove(option);
252         options.add(option);
253     }
254 
255     /**
256      * <p>
257      * Builds a list with file system options.
258      * </p>
259      * <p>
260      * <strong>Warning</strong>: the {@link #appendLocal() local file system} should be appended as last or not at all.
261      * </p>
262      * @return the list with class names.
263      * @since 2.0.0
264      */
265     public List<FileSystemOption> build() {
266         return new ArrayList<>(options);
267     }
268 }
