22  /**
23   * <p>
24   * Represents <a href="http://lesscss.org/usage/index.html#command-line-usage-options">Less compiler options</a>
25   * responsible for controlling the {@link LessCompiler} compilation process.
26   * </p>
27   * <p>
28   * Base options:
29   * </p>
30   * <ul>
31   * <li>{@link #isCompress() compress} - whether a CSS code should be compressed (default: {@code false})</li>
32   * <li>{@link #isIeCompatibility() IE compatibility} - whether a CSS code should be compatible with Internet Explorer
33   * browser (default: {@code true})</li>
34   * <li>{@link #getIncludePaths() included paths} - available include paths (default: {@code []})</li>
35   * <li>{@link #isJavaScript() JavaScript} - whether a compiler should allow usage of JavaScript language (default:
36   * {@code true})</li>
37   * <li>{@link #getLineNumbers() line numbers} - whether a compiler should generate inline source-mapping (default:
38   * {@link LineNumbersValue#OFF})</li>
39   * <li>{@link #isRelativeUrls() relative URLs} - whether a compiler should rewrite relative URLs (default:
40   * {@code false})</li>
41   * <li>{@link #getRootPath() root path} - a path which will be added to every generated import and URL in CSS code
42   * (default: {@code null})</li>
43   * <li>{@link #isSilent() silent} - whether a compiler shouldn't log compilation warnings (default: {@code false})</li>
44   * <li>{@link #isStrictImports() strict imports} - whether a compiler should disallow an @import operation inside of
45   * either @media blocks or other selector blocks (default: {@code false})</li>
46   * <li>{@link #isStrictMath() strict math} - whether a compiler should try and process all maths in Less code (default:
47   * {@code false})</li>
48   * <li>{@link #isStrictUnits() strict units} - whether a compiler should guess units in Less code when it does maths
49   * (default: {@code false})</li>
50   * </ul>
51   * <p>
52   * Source Map options:
53   * </p>
54   * <ul>
55   * <li>{@link #getSourceMapBasePath() base path} - a path that will be removed from each of the Less file paths inside
56   * the Source Map and also from the path to the map file specified in your output CSS (default: {@code null})</li>
57   * <li>{@link #isSourceMapLessInline() Less inline} - whether a compiler should include all of the Less files in to the
58   * Source Map (default: {@code false})</li>
59   * <li>{@link #getSourceMapRootPath() root path} - a path that will be prepended to each of the Less file paths inside
60   * the Source Map and also to the path to the map file specified in your output CSS (default: {@code null})</li>
61   * <li>{@link #getSourceMapUrl() URL} - a path which will overwrite the URL in the CSS that points at the Source Map
62   * file (default: {@code null})</li>
63   * </ul>
64   * <p>
65   * Non-standard options:
66   * </p>
67   * <ul>
68   * <li>{@link #getEncoding() encoding} - an encoding used to read source files and save generated code (default:
69   * {@code null} - means platform default encoding)</li>
70   * <li>{@link #getFileLoaders() file loaders} - a list with class names of the
71   * {@link biz.gabrys.lesscss.compiler2.loader.FileLoader file loaders} (default:
72   * <code>["biz.gabrys.lesscss.compiler2.loader.{@link LocalFileLoader}"]</code>)</li>
73   * </ul>
74   * @since 2.0.0
75   * @see LessOptionsBuilder
76   * @see FileLoadersOptionBuilder
77   */
78  public class LessOptions {

144         sourceMapUrl = options.sourceMapUrl;

147         fileLoaders = new ArrayList<String>(options.fileLoaders);

327     /**
328      * Checks whether a compiler should rewrite relative URLs (default: {@code false}). By default URLs are kept as-is,
329      * so if you import a file in a sub-directory that references an image, exactly the same URL will be output in the
330      * CSS code. This option allows you to re-write URLs in imported files so that the URL is always relative to the
331      * base imported file. Example:
332      * 
333      * <pre>
334      * # main.less
335      * &#64;import "files/backgrounds.less";
336      * 
337      * # files/backgrounds.less
338      * .icon-1 {
339      *     background-image: url('images/lamp-post.png');
340      * }
341      * </pre>
342      * <p>
343      * this will output the following normally
344      * </p>
345      * 
346      * <pre>
347      * .icon-1 {
348      *     background-image: url('images/lamp-post.png');
349      * }
350      * </pre>
351      * <p>
352      * but with this option on it will instead output
353      * </p>
354      * 
355      * <pre>
356      * .icon-1 {
357      *     background-image: url('files/images/lamp-post.png');
358      * }
359      * </pre>
360      * <p>
361      * You may also want to consider using the {@code data-uri} function instead of this option, which will embed images
362      * into the CSS.
363      * </p>
364      * @return {@code true} whether the compiler should rewrite relative URLs, otherwise {@code false}.
365      * @since 2.0.0
366      */
367     public boolean isRelativeUrls() {
368         return relativeUrls;
369     }

416      * Checks whether a compiler should try and process all maths in Less code (default: {@code false}). By default the
417      * compiler will try and process all maths in your Less e.g.:
418      * 
419      * <pre>
420      * .class {

422      * }
423      * </pre>

437      * </pre>
438      * <p>
439      * will be compiled to
440      * </p>
441      * 
442      * <pre>
443      * .class {
444      *     width: calc(100% - 5px);
445      *     height: 25px;
446      *     font-size: 1 / 4;
447      * }
448      * </pre>
449      * 
450      * @return {@code true} whether the compiler should try and process all maths in the Less code, otherwise

503     /**
504      * Checks whether a compiler should guess units in Less code when it does maths (default: {@code false}). By default
505      * the compiler attempts to guess at the output unit when it does maths. For instance:
506      * 
507      * <pre>
508      * .class {
509      *     property: 1px * 2px;
510      * }
511      * </pre>
512      * <p>
513      * In this case, things are clearly not right - a length multiplied by a length gives an area, but CSS does not
514      * support specifying areas. So we assume that the user meant for one of the values to be a value, not a unit of
515      * length and Less outputs 2px. With strict units on, Less assumes this is a bug in the calculation and throw an
516      * error.
517      * </p>
518      * @return {@code true} whether the compiler should guess units in the Less code when it does maths, otherwise
519      *         {@code false}.
520      * @since 2.0.0

522      */
