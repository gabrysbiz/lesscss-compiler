23  /**
24   * <p>
25   * Responsible for creating <a href="http://lesscss.org/usage/index.html#command-line-usage-options">options</a> that
26   * control the {@link NativeLessCompiler} compilation process.
27   * </p>
28   * <p>
29   * Base options:
30   * </p>
31   * <ul>
32   * <li>{@link #banner(String) banner} - a banner which will be inserted to a source file before the compilation</li>
33   * <li>{@link #compress(boolean) compression} - whether a CSS code should be compressed (default: {@code false})</li>
34   * <li>{@link #ieCompatibility(boolean) IE compatibility} - whether a CSS code should be compatible with Internet
35   * Explorer browser (default: {@code true})</li>
36   * <li>{@link #includePaths(List) included paths} - available include paths (default: {@code []})</li>
37   * <li>{@link #inputFile(String) path} - an input (source) file path</li>
38   * <li>{@link #javaScript(boolean) JavaScript} - whether a compiler should allow usage of JavaScript language (default:
39   * {@code true})</li>
40   * <li>{@link #lineNumbers(LineNumbersValue) line numbers} - whether a compiler should generate inline source-mapping
41   * (default: {@link LineNumbersValue#OFF})</li>
42   * <li>{@link #outputFile(File) output file} - an output (destination) file</li>
43   * <li>{@link #relativeUrls(boolean) relative URLs} - whether a compiler should rewrite relative URLs (default:
44   * {@code false})</li>
45   * <li>{@link #rootPath(String) root path} - a path which will be added to every generated import and URL in CSS code
46   * (default: {@code null})</li>
47   * <li>{@link #silent(boolean) silent} - whether a compiler shouldn't log compilation warnings (default:
48   * {@code false})</li>
49   * <li>{@link #strictImports(boolean) strict imports} - whether a compiler should disallow an @import operation inside
50   * of either @media blocks or other selector blocks (default: {@code false})</li>
51   * <li>{@link #strictMath(boolean) strict math} - whether a compiler should try and process all maths in Less code
52   * (default: {@code false})</li>
53   * <li>{@link #strictUnits(boolean) strict units} - whether a compiler should guess units in Less code when it does
54   * maths (default: {@code false})</li>
55   * </ul>
56   * <p>
57   * Source Map options:
58   * </p>
59   * <ul>
60   * <li>You can set where Source Map will be saved by using one of the options (mutually exclusive):
61   * <ul>
62   * <li>{@link #sourceMapDefault(boolean) default} - generate a Source Map file with a default name equal to
63   * {@link #outputFile(File) output file name} with {@code map} extension (default: {@code false})</li>
64   * <li>{@link #sourceMapFile(File) file} - generate a Source Map file (default: {@code null}).</li>
65   * <li>{@link #sourceMapInline(boolean) inline} - generate a Source Map and put it to CSS code (default: {@code false})
66   * </ul>
67   * </li>
68   * <li>{@link #sourceMapBasePath(String) base path} - a path that will be removed from each of the Less file paths
69   * inside the Source Map and also from the path to the map file specified in your output CSS (default:
70   * {@code null})</li>
71   * <li>{@link #sourceMapLessInline(boolean) Less inline} - whether a compiler should include all of the Less files in to
72   * the Source Map (default: {@code false})</li>
73   * <li>{@link #sourceMapRootPath(String) root path} - a path that will be prepended to each of the Less file paths
74   * inside the Source Map and also to the path to the map file specified in your output CSS (default: {@code null})</li>
75   * <li>{@link #sourceMapUrl(String) URL} - a path which will overwrite the URL in the CSS that points at the Source Map
76   * file (default: {@code null})</li>
77   * </ul>
78   * <p>
79   * Non-standard options:
80   * </p>
81   * <ul>
82   * <li>{@link #encoding(String) encoding} - an encoding used to read source files and save generated code (default:
83   * {@code null} - means platform default encoding)</li>
84   * <li>{@link #fileSystems(List) file systems} - a list with class names of the
85   * {@link biz.gabrys.lesscss.compiler2.filesystem.FileSystem file systems} (default:
86   * {@link LessOptions#DEFAULT_FILE_SYSTEMS})</li>
87   * </ul>
88   * <p>
89   * Example code:
90   * </p>
91   * 
92   * <pre>
93   * Collection&lt;String&gt; options = null;
94   * 
95   * // create options with an input file
96   * options = new {@link #NativeLessOptionsBuilder() NativeLessOptionsBuilder}().{@link #inputFile(String) inputFile}("/less/file.less").{@link #build() build}();
97   * 
98   * // create options with an input file and compression enabled
99   * options = new {@link #NativeLessOptionsBuilder() NativeLessOptionsBuilder}().{@link #inputFile(String) inputFile}("/less/file.less").{@link #compress(boolean) compress}(true).{@link #build() build}();
100  * 
101  * // use {@link LessOptions} object to create options 
102  * {@link LessOptions} lessOptions = new {@link LessOptions#LessOptions() LessOptions}();
103  * ... // modify options
104  * options = new {@link #NativeLessOptionsBuilder() NativeLessOptionsBuilder}().{@link #options(LessOptions) options}(lessOptions).{@link #build() build}();
105  * </pre>
106  * 
107  * @since 2.0.0
108  * @see LessOptions
109  */

357     /**
358      * Returns a command line option representation of the available include paths.
359      * @return the command line option (never {@code null}).
360      * @since 2.0.0
361      * @see #includePaths(List)
362      */
363     protected String getIncludePathsOption() {
364         final Collection<String> includePaths = options.getIncludePaths();
365         if (includePaths.isEmpty()) {
366             return "";
367         }
368         final StringBuilder paths = new StringBuilder();
369         for (final String path : includePaths) {
370             if (StringUtils.isNotBlank(path)) {
371                 final boolean separatorRequired = paths.length() != 0;
372                 if (separatorRequired) {
373                     paths.append(NativeLessCompiler.INCLUDE_PATHS_SEPARATOR);
374                 }
375                 paths.append(path.trim());
376             }
377         }
378         return "--include-path=" + paths.toString();
379     }

889     /**
890      * Returns a command line option representation of the available file systems.
891      * @return the command line option (never {@code null}).
892      * @since 2.0.0
893      * @see #fileSystems(List)
894      */
895     protected String getFileSystemsOption() {
896         final Collection<String> fileSystems = options.getFileSystems();
897         if (fileSystems.isEmpty() || LessOptions.DEFAULT_FILE_SYSTEMS.equals(fileSystems)) {
898             return "";
899         }
900         final StringBuilder systems = new StringBuilder();
901         for (final String system : fileSystems) {
902             if (StringUtils.isNotBlank(system)) {
903                 final boolean separatorRequired = systems.length() != 0;
904                 if (separatorRequired) {
905                     systems.append(',');
906                 }
907                 systems.append(system.trim());
908             }
909         }
910         return "--file-systems=" + systems.toString();
911     }

939     /**
940      * Builds a collection with configuration options for the {@link NativeLessCompiler} compilation process.
941      * @return the collection with configuration options.
942      * @throws BuilderCreationException if you set an output file without setting an input file.
943      * @since 2.0.0
944      */
945     public Collection<String> build() {
946         final ListWithoutEmptyValuesBuilder<String> configurationOptions = new ListWithoutEmptyValuesBuilder<String>(new Checker<String>() {
947 
948             @Override
949             public boolean isNotEmpty(final String value) {
950                 return StringUtils.isNotBlank(value);
951             }
952         });
953         configurationOptions.append(getSilentOption());
954         configurationOptions.append(getStrictImportsOption());
955         configurationOptions.append(getCompressOption());
956         configurationOptions.append(getIeCompatibilityOption());
957         configurationOptions.append(getJavaScriptOption());
958         configurationOptions.append(getIncludePathsOption());
959         configurationOptions.append(getLineNumbersOption());
960 
961         configurationOptions.append(getSourceMapDefaultOption());
962         configurationOptions.append(getSourceMapFileOption());
963         configurationOptions.append(getSourceMapInlineOptions());
964         configurationOptions.append(getSourceMapLessInlineOption());
965         configurationOptions.append(getSourceMapRootPathOption());
966         configurationOptions.append(getSourceMapBasePathOption());
967         configurationOptions.append(getSourceMapUrlOption());
968 
969         configurationOptions.append(getRootPathOption());
970         configurationOptions.append(getRelativeUrlsOption());
971         configurationOptions.append(getStrictMathOption());
972         configurationOptions.append(getStrictUnitsOption());
973 
974         configurationOptions.append(getEncodingOption());
975         configurationOptions.append(getFileSystemsOption());
976 
977         configurationOptions.append(getBannerOption());
978 
979         final String inputPath = getInputFileOption();
980         configurationOptions.append(inputPath);
981         final String outputPath = getOutputFileOption();
982         if (StringUtils.isNotBlank(outputPath)) {
983             if (StringUtils.isBlank(inputPath)) {
984                 throw new BuilderCreationException("Input file is required when otuput file is set (see inputFile(String) method)");
985             }
986             configurationOptions.append(outputPath);
987         }
988 
989         return configurationOptions.build();
