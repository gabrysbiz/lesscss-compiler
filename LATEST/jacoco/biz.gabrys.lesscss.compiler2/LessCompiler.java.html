package biz.gabrys.lesscss.compiler2;

import java.io.File;
import java.io.IOException;

import biz.gabrys.lesscss.compiler2.util.io.IOUtils;
import biz.gabrys.lesscss.compiler2.util.io.TemporaryFileFactory;
import biz.gabrys.lesscss.compiler2.util.io.SystemDefaultTemporaryFileFactory;

/**
 * <p>
 * <a href="http://lesscss.org/">Less</a> compiler compatible with
 * <a href="https://github.com/less/less.js/releases/tag/v1.7.5">Less 1.7.5</a>.
 * </p>
 * <p>
 * How to use:
 * </p>
 * 
 * <pre>
 * // create compiler and source file
 * LessCompiler compiler = new LessCompiler();
 * File source = new File("/source.less");
 * 
 * // compile source file
 * String cssCode = compiler.compile(source);
 * 
 * // compile source file and save to output file
 * compiler.compile(source, new File("/output.css"));
 * 
 * // compile source file and compress CSS code
 * String compressedCssCode = compiler.compileAndCompress(source);
 * </pre>
 * 
 * @since 2.0.0
 */
public class LessCompiler {

    private final NativeLessCompiler compiler;
    private final TemporaryFileFactory fileFactory;

    /**
     * Constructs a new instance.
     * @since 2.0.0
     */
    public LessCompiler() {
        this(new NativeLessCompiler());
    }

    /**
     * Constructs a new instance.
     * @param compiler the native compiler used to compile source files.
     * @since 2.0.0
     */
    public LessCompiler(final NativeLessCompiler compiler) {
        this(compiler, new SystemDefaultTemporaryFileFactory());
    }

    /**
     * Constructs a new instance.
     * @param fileFactory the file factory used to create source files when Less code is passed by string parameter.
     * @since 2.0.0
     */
    public LessCompiler(final TemporaryFileFactory fileFactory) {
        this(new NativeLessCompiler(), fileFactory);
    }

    /**
     * Constructs a new instance.
     * @param compiler the native compiler used to compile source files.
     * @param fileFactory the file factory used to create source files when Less code is passed by string parameter.
     * @since 2.0.0
     */
    public LessCompiler(final NativeLessCompiler compiler, final TemporaryFileFactory fileFactory) {
        this.compiler = compiler;
        this.fileFactory = fileFactory;
    }

    /**
     * Compiles a Less source file and returns a CSS code.
     * @param source the Less code (cannot be {@code null}).
     * @return the CSS code.
     * @throws IllegalArgumentException if the Less code is {@code null}.
     * @throws InitializationException if an error occurred during compiler initialization.
     * @throws ConfigurationException if the compiler is configured incorrectly.
     * @throws ReadFileException if any &#64;import operation point to non-existent file.
     * @throws SyntaxException if a syntax error occurred during source file compilation.
     * @throws CompilerException if an other error occurred during execution.
     * @since 2.0.0
     */
    public String compile(final CharSequence source) {
        validateSourceString(source);
        final File sourceFile = createTemporaryFileWithCode(source);
        final NativeLessCompilerOptionsBuilder builder = createOptionsBuilder();
        builder.withInputFile(sourceFile);
        final String css = compiler.execute(builder.create());
        deleteFile(sourceFile);
        return css;
    }

    /**
     * Compiles a Less source file and returns a CSS code.
     * @param source the source file (cannot be {@code null} and must exist).
     * @return the CSS code.
     * @throws IllegalArgumentException if the source file is {@code null} or does not exist.
     * @throws InitializationException if an error occurred during compiler initialization.
     * @throws ConfigurationException if the compiler is configured incorrectly.
     * @throws ReadFileException if the compiler cannot read source file or any &#64;import operation point to
     *             non-existent file.
     * @throws SyntaxException if a syntax error occurred during source file compilation.
     * @throws CompilerException if an other error occurred during execution.
     * @since 2.0.0
     */
    public String compile(final File source) {
        validateSourceFile(source);
        final NativeLessCompilerOptionsBuilder builder = createOptionsBuilder();
        builder.withInputFile(source);
        return compiler.execute(builder.create());
    }

    /**
     * Compiles a Less source file and saves CSS code in an output file.
     * @param source the source file (cannot be {@code null} and must exist).
     * @param output the output file (cannot be {@code null}).
     * @throws IllegalArgumentException if the source file is {@code null} or does not exist or the output file is
     *             {@code null}.
     * @throws InitializationException if an error occurred during compiler initialization.
     * @throws ConfigurationException if the compiler is configured incorrectly.
     * @throws ReadFileException if the compiler cannot read source file or any &#64;import operation point to
     *             non-existent file.
     * @throws SyntaxException if a syntax error occurred during source file compilation.
     * @throws CompilerException if an other error occurred during execution.
     * @since 2.0.0
     */
    public void compile(final File source, final File output) {
        validateSourceFile(source);
        validateOutputFile(output);
        final NativeLessCompilerOptionsBuilder builder = createOptionsBuilder();
        builder.withInputFile(source).withOutputFile(output);
        compiler.execute(builder.create());
    }

    /**
     * Compiles a Less source file with custom configuration options and returns a CSS code.
     * @param source the Less code (cannot be {@code null}).
     * @param options the configuration options (can be {@code null}).
     * @return the CSS code.
     * @throws IllegalArgumentException if the Less code is {@code null}.
     * @throws InitializationException if an error occurred during compiler initialization.
     * @throws ConfigurationException if the compiler is configured incorrectly.
     * @throws ReadFileException if any &#64;import operation point to non-existent file.
     * @throws SyntaxException if a syntax error occurred during source file compilation.
     * @throws CompilerException if an other error occurred during execution.
     * @since 2.0.0
     */
    public String compile(final CharSequence source, final LessCompilerOptions options) {
        validateSourceString(source);
        final File sourceFile = createTemporaryFileWithCode(source);
        final NativeLessCompilerOptionsBuilder builder = createOptionsBuilder();
        builder.withInputFile(sourceFile).withOptions(options);
        final String css = compiler.execute(builder.create());
        deleteFile(sourceFile);
        return css;
    }

    /**
     * Compiles a Less source file with custom configuration options to a CSS code.
     * @param source the source file (cannot be {@code null} and must exist).
     * @param options the configuration options (can be {@code null}).
     * @return the CSS code.
     * @throws IllegalArgumentException if the source file is {@code null} or does not exist.
     * @throws InitializationException if an error occurred during compiler initialization.
     * @throws ConfigurationException if the compiler is configured incorrectly.
     * @throws ReadFileException if the compiler cannot read source file or any &#64;import operation point to
     *             non-existent file.
     * @throws SyntaxException if a syntax error occurred during source file compilation.
     * @throws CompilerException if an other error occurred during execution.
     * @since 2.0.0
     */
    public String compile(final File source, final LessCompilerOptions options) {
        validateSourceFile(source);
        final NativeLessCompilerOptionsBuilder builder = createOptionsBuilder();
        builder.withInputFile(source).withOptions(options);
        return compiler.execute(builder.create());
    }

    /**
     * Compiles a Less source file with custom configuration options and saves result in an output file.
     * @param source the source file (cannot be {@code null} and must exist).
     * @param output the output file (cannot be {@code null}).
     * @param options the configuration options (can be {@code null}).
     * @throws IllegalArgumentException if the source file is {@code null} or does not exist ot the output file is
     *             {@code null}.
     * @throws InitializationException if an error occurred during compiler initialization.
     * @throws ConfigurationException if the compiler is configured incorrectly.
     * @throws ReadFileException if the compiler cannot read source file or any &#64;import operation point to
     *             non-existent file.
     * @throws SyntaxException if a syntax error occurred during source file compilation.
     * @throws CompilerException if an other error occurred during execution.
     * @since 2.0.0
     */
    public void compile(final File source, final File output, final LessCompilerOptions options) {
        validateSourceFile(source);
        validateOutputFile(output);
        final NativeLessCompilerOptionsBuilder builder = createOptionsBuilder();
        builder.withInputFile(source).withOutputFile(output).withOptions(options);
        compiler.execute(builder.create());
    }

    /**
     * Compiles a Less source file and returns a compressed CSS code.
     * @param source the Less code (cannot be {@code null}).
     * @return the compressed CSS code.
     * @throws IllegalArgumentException if the Less code is {@code null}.
     * @throws InitializationException if an error occurred during compiler initialization.
     * @throws ConfigurationException if the compiler is configured incorrectly.
     * @throws ReadFileException if any &#64;import operation point to non-existent file.
     * @throws SyntaxException if a syntax error occurred during source file compilation.
     * @throws CompilerException if an other error occurred during execution.
     * @since 2.0.0
     */
    public String compileAndCompress(final CharSequence source) {
        validateSourceString(source);
        final File sourceFile = createTemporaryFileWithCode(source);
        final NativeLessCompilerOptionsBuilder builder = createOptionsBuilder();
        builder.withInputFile(sourceFile).withCompress(true);
        final String css = compiler.execute(builder.create());
        deleteFile(sourceFile);
        return css;
    }

    /**
     * Compiles a Less source file to a compressed CSS code.
     * @param source the source file (cannot be {@code null} and must exist).
     * @return the compressed CSS code.
     * @throws IllegalArgumentException if the source file is {@code null} or does not exist.
     * @throws InitializationException if an error occurred during compiler initialization.
     * @throws ConfigurationException if the compiler is configured incorrectly.
     * @throws ReadFileException if the compiler cannot read source file or any &#64;import operation point to
     *             non-existent file.
     * @throws SyntaxException if a syntax error occurred during source file compilation.
     * @throws CompilerException if an other error occurred during execution.
     * @since 2.0.0
     */
    public String compileAndCompress(final File source) {
        validateSourceFile(source);
        final NativeLessCompilerOptionsBuilder builder = createOptionsBuilder();
        builder.withInputFile(source).withCompress(true);
        return compiler.execute(builder.create());
    }

    /**
     * Compiles a Less source file to a compressed CSS code and saves it in an output file.
     * @param source the source file (cannot be {@code null} and must exist).
     * @param output the output file (cannot be {@code null}).
     * @throws IllegalArgumentException if the source file is {@code null} or does not exist or the output file is
     *             {@code null}.
     * @throws InitializationException if an error occurred during compiler initialization.
     * @throws ConfigurationException if the compiler is configured incorrectly.
     * @throws ReadFileException if the compiler cannot read source file or any &#64;import operation point to
     *             non-existent file.
     * @throws SyntaxException if a syntax error occurred during source file compilation.
     * @throws CompilerException if an other error occurred during execution.
     * @since 2.0.0
     */
    public void compileAndCompress(final File source, final File output) {
        validateSourceFile(source);
        validateOutputFile(output);
        final NativeLessCompilerOptionsBuilder builder = createOptionsBuilder();
        builder.withInputFile(source).withOutputFile(output).withCompress(true);
        compiler.execute(builder.create());
    }

    /**
     * Compiles a Less source file to a CSS code with inline Source Map.
     * @param source the source file (cannot be {@code null} and must exist).
     * @return the CSS code with inline Source Map.
     * @throws IllegalArgumentException if the source file is {@code null} or does not exist.
     * @throws InitializationException if an error occurred during compiler initialization.
     * @throws ConfigurationException if the compiler is configured incorrectly.
     * @throws ReadFileException if the compiler cannot read source file or any &#64;import operation point to
     *             non-existent file.
     * @throws SyntaxException if a syntax error occurred during source file compilation.
     * @throws CompilerException if an other error occurred during execution.
     * @since 2.0.0
     */
    public String compileWithInlineSourceMap(final File source) {
        validateSourceFile(source);
        final NativeLessCompilerOptionsBuilder builder = createOptionsBuilder();
        builder.withInputFile(source).withSourceMapInline(true);
        return compiler.execute(builder.create());
    }

    /**
     * Compiles a Less source file to a CSS code with inline Source Map and saves it in an output file.
     * @param source the source file (cannot be {@code null} and must exist).
     * @param output the output file (cannot be {@code null}).
     * @return the CSS code with inline Source Map.
     * @throws IllegalArgumentException if the source file is {@code null} or does not exist or the output file is
     *             {@code null}.
     * @throws InitializationException if an error occurred during compiler initialization.
     * @throws ConfigurationException if the compiler is configured incorrectly.
     * @throws ReadFileException if the compiler cannot read source file or any &#64;import operation point to
     *             non-existent file.
     * @throws SyntaxException if a syntax error occurred during source file compilation.
     * @throws CompilerException if an other error occurred during execution.
     * @since 2.0.0
     */
    public String compileWithInlineSourceMap(final File source, final File output) {
        validateSourceFile(source);
        validateOutputFile(output);
        final NativeLessCompilerOptionsBuilder builder = createOptionsBuilder();
        builder.withInputFile(source).withOutputFile(output).withSourceMapInline(true);
        return compiler.execute(builder.create());
    }

    /**
     * Compiles a Less source file to a compressed CSS code with inline Source Map.
     * @param source the source file (cannot be {@code null} and must exist).
     * @return the compressed CSS code with inline Source Map.
     * @throws IllegalArgumentException if the source file is {@code null} or does not exist.
     * @throws InitializationException if an error occurred during compiler initialization.
     * @throws ConfigurationException if the compiler is configured incorrectly.
     * @throws ReadFileException if the compiler cannot read source file or any &#64;import operation point to
     *             non-existent file.
     * @throws SyntaxException if a syntax error occurred during source file compilation.
     * @throws CompilerException if an other error occurred during execution.
     * @since 2.0.0
     */
    public String compileWithInlineSourceMapAndCompress(final File source) {
        validateSourceFile(source);
        final NativeLessCompilerOptionsBuilder builder = createOptionsBuilder();
        builder.withInputFile(source).withSourceMapInline(true).withCompress(true);
        return compiler.execute(builder.create());
    }

    /**
     * Compiles a Less source file to a compressed CSS code with inline Source Map and saves it in an output file.
     * @param source the source file (cannot be {@code null} and must exist).
     * @param output the output file (cannot be {@code null}).
     * @return the compressed CSS code with inline Source Map.
     * @throws IllegalArgumentException if the source file is {@code null} or does not exist or the output file is
     *             {@code null}.
     * @throws InitializationException if an error occurred during compiler initialization.
     * @throws ConfigurationException if the compiler is configured incorrectly.
     * @throws ReadFileException if the compiler cannot read source file or any &#64;import operation point to
     *             non-existent file.
     * @throws SyntaxException if a syntax error occurred during source file compilation.
     * @throws CompilerException if an other error occurred during execution.
     * @since 2.0.0
     */
    public String compileWithInlineSourceMapAndCompress(final File source, final File output) {
        validateSourceFile(source);
        validateOutputFile(output);
        final NativeLessCompilerOptionsBuilder builder = createOptionsBuilder();
        builder.withInputFile(source).withOutputFile(output).withSourceMapInline(true).withCompress(true);
        return compiler.execute(builder.create());
    }

    /**
     * Compiles a Less source file to a CSS code with full inline Source Map. Full means that include the compiler will
     * include all of the Less files in to the Source Map.
     * @param source the source file (cannot be {@code null} and must exist).
     * @return the CSS code with inline Source Map.
     * @throws IllegalArgumentException if the source file is {@code null} or does not exist.
     * @throws InitializationException if an error occurred during compiler initialization.
     * @throws ConfigurationException if the compiler is configured incorrectly.
     * @throws ReadFileException if the compiler cannot read source file or any &#64;import operation point to
     *             non-existent file.
     * @throws SyntaxException if a syntax error occurred during source file compilation.
     * @throws CompilerException if an other error occurred during execution.
     * @since 2.0.0
     */
    public String compileWithFullInlineSourceMap(final File source) {
        validateSourceFile(source);
        final NativeLessCompilerOptionsBuilder builder = createOptionsBuilder();
        builder.withInputFile(source).withSourceMapInline(true).withSourceMapLessInline(true);
        return compiler.execute(builder.create());
    }

    /**
     * Compiles a Less source file to a CSS code with full inline Source Map and saves it in an output file. Full means
     * that include the compiler will include all of the Less files in to the Source Map.
     * @param source the source file (cannot be {@code null} and must exist).
     * @param output the output file (cannot be {@code null}).
     * @return the CSS code with inline Source Map.
     * @throws IllegalArgumentException if the source file is {@code null} or does not exist or the output file is
     *             {@code null}.
     * @throws InitializationException if an error occurred during compiler initialization.
     * @throws ConfigurationException if the compiler is configured incorrectly.
     * @throws ReadFileException if the compiler cannot read source file or any &#64;import operation point to
     *             non-existent file.
     * @throws SyntaxException if a syntax error occurred during source file compilation.
     * @throws CompilerException if an other error occurred during execution.
     * @since 2.0.0
     */
    public String compileWithFullInlineSourceMap(final File source, final File output) {
        validateSourceFile(source);
        validateOutputFile(output);
        final NativeLessCompilerOptionsBuilder builder = createOptionsBuilder();
        builder.withInputFile(source).withOutputFile(output).withSourceMapInline(true).withSourceMapLessInline(true);
        return compiler.execute(builder.create());
    }

    /**
     * Validates a source text.
     * @param source the source text.
     * @throws IllegalArgumentException if the source text is {@code null}.
     * @since 2.0.0
     */
    protected void validateSourceString(final CharSequence source) {
        if (source == null) {
            throw new IllegalArgumentException("Source text cannot be null");
        }
    }

    /**
     * Validates source file.
     * @param source the source file.
     * @throws IllegalArgumentException if the source file is {@code null} or does not exist.
     * @since 2.0.0
     */
    protected void validateSourceFile(final File source) {
        if (source == null) {
            throw new IllegalArgumentException("Source file cannot be null");
        }
        if (!source.exists()) {
            throw new IllegalArgumentException("Source file does not exist: " + source.getAbsolutePath());
        }
    }

    /**
     * Validates output file.
     * @param output the output file.
     * @throws IllegalArgumentException if the output file is {@code null}.
     * @since 2.0.0
     */
    protected void validateOutputFile(final File output) {
        if (output == null) {
            throw new IllegalArgumentException("Output file cannot be null");
        }
    }

    /**
     * Creates a new instance of the {@link NativeLessCompilerOptionsBuilder} used to create configuration options for
     * the native compiler.
     * @return the new instance of the {@link NativeLessCompilerOptionsBuilder}.
     * @since 2.0.0
     */
    protected NativeLessCompilerOptionsBuilder createOptionsBuilder() {
        return new NativeLessCompilerOptionsBuilder();
    }

    /**
     * Creates a temporary file.
     * @param content the file content (cannot be {@code null}).
     * @return the temporary file.
     * @throws IllegalArgumentException if the file or the content is {@code null}.
     * @throws CompilerException if cannot create the temporary file.
     * @since 2.0.0
     */
    protected File createTemporaryFileWithCode(final CharSequence content) {
        File file = null;
        try {
            file = fileFactory.create();
        } catch (final IOException e) {
            throw new CompilerException("Cannot create temporary file", e);
        }
        try {
            IOUtils.write(file, content);
        } catch (final IOException e) {
            throw new CompilerException("Cannot create temporary file with Less source code", e);
        }
        return file;
    }

    /**
     * Deletes a file.
     * @param file the file (cannot be {@code null}).
     * @throws IllegalArgumentException if the file is {@code null}.
     * @since 2.0.0
     */
    protected void deleteFile(final File file) {
        if (file == null) {
            throw new IllegalArgumentException("File cannot be null");
        }
        file.delete();
    }
}
