package biz.gabrys.lesscss.compiler2;

import java.util.ArrayList;
import java.util.LinkedHashSet;
import java.util.List;
import java.util.Set;

import biz.gabrys.lesscss.compiler2.loader.ClassPathFileLoader;
import biz.gabrys.lesscss.compiler2.loader.FileLoader;
import biz.gabrys.lesscss.compiler2.loader.FtpFileLoader;
import biz.gabrys.lesscss.compiler2.loader.HttpFileLoader;
import biz.gabrys.lesscss.compiler2.loader.LocalFileLoader;

/**
 * <p>
 * Allows to create value of the {@link LessOptions#getFileLoaders() file loaders} option.
 * </p>
 * <p>
 * Example code:
 * </p>
 * 
 * <pre>
 * List&lt;String&gt; fileLoaders;
 * // create a list with standard loaders
 * fileLoaders = new {@link #FileLoadersOptionBuilder() FileLoadersOptionBuilder}().{@link #withHttp() withHttp}().{@link #withLocal() withLocal}().{@link #build() build}();
 * 
 * // create a list with standard loaders (short version)
 * fileLoaders = new {@link #FileLoadersOptionBuilder() FileLoadersOptionBuilder}().{@link #withStandard() withStandard}().{@link #build() build}();
 * 
 * // create a list with custom loader
 * class CustomFileLoader implements {@link FileLoader} {
 *      // omitted implementation
 * }
 * 
 * fileLoaders = new {@link #FileLoadersOptionBuilder() FileLoadersOptionBuilder}().{@link #withCustom(Class) withCustom}(CustomFileLoader.class).{@link #build() build}();
 * </pre>
 * 
 * @since 2.0.0
 * @see FileLoader
 * @see LessOptions
 */
public class FileLoadersOptionBuilder {

    private final Set<String> fileLoaders = new LinkedHashSet<String>();

    /**
     * Constructs a new instance.
     * @since 2.0.0
     */
    public FileLoadersOptionBuilder() {
        // do nothing
    }

    /**
     * Appends standard file loaders at the end or does nothing if you try to add the same file loaders again.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see HttpFileLoader
     * @see LocalFileLoader
     */
    public FileLoadersOptionBuilder withStandard() {
        return withHttp().withLocal();
    }

    /**
     * Appends the {@link LocalFileLoader} at the end or does nothing if you try to add the same file loader again.
     * @return {@code this} builder.
     * @since 2.0.0
     */
    public FileLoadersOptionBuilder withLocal() {
        fileLoaders.add(LocalFileLoader.class.getName());
        return this;
    }

    /**
     * Appends the {@link HttpFileLoader} at the end or does nothing if you try to add the same file loader again.
     * @return {@code this} builder.
     * @since 2.0.0
     */
    public FileLoadersOptionBuilder withHttp() {
        fileLoaders.add(HttpFileLoader.class.getName());
        return this;
    }

    /**
     * Appends the {@link ClassPathFileLoader} at the end or does nothing if you try to add the same file loader again.
     * @return {@code this} builder.
     * @since 2.0.0
     */
    public FileLoadersOptionBuilder withClassPath() {
        fileLoaders.add(ClassPathFileLoader.class.getName());
        return this;
    }

    /**
     * <p>
     * Appends the {@link FtpFileLoader} at the end or does nothing if you try to add the same file loader again.
     * </p>
     * <p>
     * <strong>Warning</strong>: requires <a href="https://commons.apache.org/proper/commons-net/">Apache Commons
     * Net</a> library in the classpath.
     * </p>
     * @return {@code this} builder.
     * @since 2.0.0
     */
    public FileLoadersOptionBuilder withFtp() {
        fileLoaders.add(FtpFileLoader.class.getName());
        return this;
    }

    /**
     * Appends a {@link FileLoader custom file loader implementation} at the end or does nothing if you try to add the
     * same file loader again.
     * @param clazz the custom file loader implementation class.
     * @return {@code this} builder.
     * @throws IllegalArgumentException is clazz is equal to {@code null}.
     * @since 2.0.0
     */
    public FileLoadersOptionBuilder withCustom(final Class<? extends FileLoader> clazz) {
        if (clazz == null) {
            throw new IllegalArgumentException("Clazz cannot be null");
        }
        fileLoaders.add(clazz.getName());
        return this;
    }

    /**
     * Builds a list with {@link FileLoader file loaders} class names.
     * @return the list with class names.
     * @since 2.0.0
     */
    public List<String> build() {
        return new ArrayList<String>(fileLoaders);
    }
}
