package biz.gabrys.lesscss.compiler2;

import java.util.Arrays;
import java.util.List;

import biz.gabrys.lesscss.compiler2.LessCompilerOptions.LineNumbersValue;

/**
 * Responsible for creating new instances of the {@link LessCompilerOptions}.
 * @since 2.0.0
 */
public class LessCompilerOptionsBuilder {

    LessCompilerOptions options;

    /**
     * Constructs a new instance.
     * @since 2.0.0
     */
    public LessCompilerOptionsBuilder() {
        options = new LessCompilerOptions();
    }

    /**
     * Constructs a new instance and copies.
     * @param options the initial builder state.
     * @since 2.0.0
     */
    public LessCompilerOptionsBuilder(final LessCompilerOptions options) {
        if (options == null) {
            throw new IllegalArgumentException("Options cannot be null");
        }
        this.options = new LessCompilerOptions(options);
    }

    /**
     * Sets whether a compiler shouldn't log compilation warnings.
     * @param silent {@code true} whether the compiler shouldn't log compilation warnings, otherwise {@code false}.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #silentOn()
     * @see #silentOff()
     */
    public LessCompilerOptionsBuilder silent(final boolean silent) {
        options.setSilent(silent);
        return this;
    }

    /**
     * Disables logging compilation warnings.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #silent(boolean)
     * @see #silentOff()
     */
    public LessCompilerOptionsBuilder silentOn() {
        return silent(true);
    }

    /**
     * Enables logging compilation warnings.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #silent(boolean)
     * @see #silentOn()
     */
    public LessCompilerOptionsBuilder silentOff() {
        return silent(false);
    }

    /**
     * Sets whether a compiler should disallow an &#64;import operation inside of either &#64;media blocks or other
     * selector blocks.
     * @param strictImports {@code true} whether the compiler should disallow the &#64;import operation inside of either
     *            &#64;media blocks or other selector blocks, otherwise {@code false}.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #strictImportsOn()
     * @see #strictImportsOff()
     */
    public LessCompilerOptionsBuilder strictImports(final boolean strictImports) {
        options.setStrictImports(strictImports);
        return this;
    }

    /**
     * Disallows usage of an &#64;import operation inside of either &#64;media blocks or other selector blocks.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #strictImports(boolean)
     * @see #strictImportsOff()
     */
    public LessCompilerOptionsBuilder strictImportsOn() {
        return strictImports(true);
    }

    /**
     * Allows usage of an &#64;import operation inside of either &#64;media blocks or other selector blocks.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #strictImports(boolean)
     * @see #strictImportsOn()
     */
    public LessCompilerOptionsBuilder strictImportsOff() {
        return strictImports(false);
    }

    /**
     * Sets whether a CSS code should be compressed.
     * @param compress {@code true} whether the CSS code should be compressed, otherwise {@code false}.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #compressOn()
     * @see #compressOff()
     */
    public LessCompilerOptionsBuilder compress(final boolean compress) {
        options.setCompress(compress);
        return this;
    }

    /**
     * Enables a CSS code compression.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #compress(boolean)
     * @see #compressOff()
     */
    public LessCompilerOptionsBuilder compressOn() {
        return compress(true);
    }

    /**
     * Disables a CSS code compression.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #compress(boolean)
     * @see #compressOn()
     */
    public LessCompilerOptionsBuilder compressOff() {
        return compress(false);
    }

    /**
     * Sets whether a CSS code should be compatible with IE browser.
     * @param ieCompatibility {@code true} whether the CSS code should be compatible with IE browser, otherwise
     *            {@code false}.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #ieCompatabilityOn()
     * @see #ieCompatabilityOff()
     */
    public LessCompilerOptionsBuilder ieCompatability(final boolean ieCompatibility) {
        options.setIeCompatibility(ieCompatibility);
        return this;
    }

    /**
     * Sets that a CSS code should be compatible with IE browser.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #ieCompatability(boolean)
     * @see #ieCompatabilityOff()
     */
    public LessCompilerOptionsBuilder ieCompatabilityOn() {
        return ieCompatability(true);
    }

    /**
     * Sets that a CSS code does not have to be compatible with IE browser.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #ieCompatability(boolean)
     * @see #ieCompatabilityOn()
     */
    public LessCompilerOptionsBuilder ieCompatabilityOff() {
        return ieCompatability(false);
    }

    /**
     * Sets whether a compiler should allow usage of JavaScript language.
     * @param javaScript {@code true} whether the compiler should allow usage of JavaScript language, otherwise
     *            {@code false}.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #javaScriptOn()
     * @see #javaScriptOff()
     */
    public LessCompilerOptionsBuilder javaScript(final boolean javaScript) {
        options.setJavaScript(javaScript);
        return this;
    }

    /**
     * Allows usage of JavaScript language.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #javaScript(boolean)
     * @see #javaScriptOff()
     */
    public LessCompilerOptionsBuilder javaScriptOn() {
        return javaScript(true);
    }

    /**
     * Disallows usage of JavaScript language.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #javaScript(boolean)
     * @see #javaScriptOn()
     */
    public LessCompilerOptionsBuilder javaScriptOff() {
        return javaScript(false);
    }

    /**
     * Sets available include paths.
     * @param includePaths the available include paths ({@code null} is treated as empty collection).
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #includePaths(String[])
     * @see #includePathsNull()
     */
    public LessCompilerOptionsBuilder includePaths(final List<String> includePaths) {
        options.setIncludePaths(includePaths);
        return this;
    }

    /**
     * Sets available include paths.
     * @param includePaths the available include paths ({@code null} is treated as empty array).
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #includePaths(List)
     * @see #includePathsNull()
     */
    public LessCompilerOptionsBuilder includePaths(final String[] includePaths) {
        if (includePaths == null) {
            options.setIncludePaths(null);
        } else {
            options.setIncludePaths(Arrays.asList(includePaths));
        }
        return this;
    }

    /**
     * Clears available include paths.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #includePaths(List)
     * @see #includePaths(String[])
     */
    public LessCompilerOptionsBuilder includePathsNull() {
        options.setIncludePaths(null);
        return this;
    }

    /**
     * Sets whether a compiler should generate inline source-mapping.
     * @param lineNumbers the line numbers value ({@code null} is treated as {@link LineNumbersValue#OFF}).
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #lineNumbersOff()
     * @see #lineNumbersComments()
     * @see #lineNumbersMediaQuery()
     * @see #lineNumbersAll()
     */
    public LessCompilerOptionsBuilder lineNumbers(final LineNumbersValue lineNumbers) {
        options.setLineNumbers(lineNumbers);
        return this;
    }

    /**
     * Disables generating inline source-mapping.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #lineNumbers(LessCompilerOptions.LineNumbersValue)
     * @see #lineNumbersComments()
     * @see #lineNumbersMediaQuery()
     * @see #lineNumbersAll()
     */
    public LessCompilerOptionsBuilder lineNumbersOff() {
        return lineNumbers(LineNumbersValue.OFF);
    }

    /**
     * Enables generating inline source-mapping in comments blocks.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #lineNumbers(LessCompilerOptions.LineNumbersValue)
     * @see #lineNumbersOff()
     * @see #lineNumbersMediaQuery()
     * @see #lineNumbersAll()
     */
    public LessCompilerOptionsBuilder lineNumbersComments() {
        return lineNumbers(LineNumbersValue.COMMENTS);
    }

    /**
     * Enables generating inline source-mapping in &#64;media queries.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #lineNumbers(LessCompilerOptions.LineNumbersValue)
     * @see #lineNumbersOff()
     * @see #lineNumbersComments()
     * @see #lineNumbersAll()
     */
    public LessCompilerOptionsBuilder lineNumbersMediaQuery() {
        return lineNumbers(LineNumbersValue.MEDIA_QUERY);
    }

    /**
     * Enables generating inline source-mapping in comments blocks and &#64;media queries.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #lineNumbers(LessCompilerOptions.LineNumbersValue)
     * @see #lineNumbersOff()
     * @see #lineNumbersComments()
     * @see #lineNumbersMediaQuery()
     */
    public LessCompilerOptionsBuilder lineNumbersAll() {
        return lineNumbers(LineNumbersValue.ALL);
    }

    /**
     * Sets a path which will be added to every generated import and URL in CSS code.
     * @param rootPath the root path.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #rootPathNull()
     */
    public LessCompilerOptionsBuilder rootPath(final String rootPath) {
        options.setRootPath(rootPath);
        return this;
    }

    /**
     * Clears a value of the root path option.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #rootPath(String)
     */
    public LessCompilerOptionsBuilder rootPathNull() {
        return rootPath(null);
    }

    /**
     * Sets whether a compiler should rewrite relative URLs.
     * @param relativeUrls {@code true} whether the compiler should rewrite relative URLs, otherwise {@code false}.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #relativeUrlsOn()
     * @see #relativeUrlsOff()
     */
    public LessCompilerOptionsBuilder relativeUrls(final boolean relativeUrls) {
        options.setRelativeUrls(relativeUrls);
        return this;
    }

    /**
     * Enables rewriting relative URLs.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #relativeUrls(boolean)
     * @see #relativeUrlsOff()
     */
    public LessCompilerOptionsBuilder relativeUrlsOn() {
        return relativeUrls(true);
    }

    /**
     * Disables rewriting relative URLs.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #relativeUrls(boolean)
     * @see #relativeUrlsOn()
     */
    public LessCompilerOptionsBuilder relativeUrlsOff() {
        return relativeUrls(false);
    }

    /**
     * Sets whether a compiler should try and process all maths in Less code.
     * @param strictMath {@code true} whether the compiler should try and process all maths in the Less code, otherwise
     *            {@code false}.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #strictMathOn()
     * @see #strictMathOff()
     */
    public LessCompilerOptionsBuilder strictMath(final boolean strictMath) {
        options.setStrictMath(strictMath);
        return this;
    }

    /**
     * Sets that a compiler should try and process all maths in Less code.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #strictMath(boolean)
     * @see #strictMathOff()
     */
    public LessCompilerOptionsBuilder strictMathOn() {
        return strictMath(true);
    }

    /**
     * Sets that a compiler shouldn't try and process all maths in Less code.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #strictMath(boolean)
     * @see #strictMathOn()
     */
    public LessCompilerOptionsBuilder strictMathOff() {
        return strictMath(false);
    }

    /**
     * Sets whether a compiler should guess units in Less code when it does maths.
     * @param strictUnits {@code true} whether the compiler should guess units in the Less code when it does maths,
     *            otherwise {@code false}.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #strictUnitsOn()
     * @see #strictUnitsOff()
     */
    public LessCompilerOptionsBuilder strictUnits(final boolean strictUnits) {
        options.setStrictUnits(strictUnits);
        return this;
    }

    /**
     * Sets that a compiler should guess units in Less code when it does maths.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #strictUnits(boolean)
     * @see #strictUnitsOff()
     */
    public LessCompilerOptionsBuilder strictUnitsOn() {
        return strictUnits(true);
    }

    /**
     * Sets that a compiler shouldn't guess units in Less code when it does maths.
     * @return {@code this} builder.
     * @since 2.0.0
     * @see #strictUnits(boolean)
     * @see #strictUnitsOn()
     */
    public LessCompilerOptionsBuilder strictUnitsOff() {
        return strictUnits(false);
    }

    /**
     * Builds a new instance of the {@link LessCompilerOptions} and resets {@code this} builder state.
     * @return the new instance of the {@link LessCompilerOptions}.
     * @since 2.0.0
     */
    public LessCompilerOptions build() {
        final LessCompilerOptions result = options;
        options = new LessCompilerOptions();
        return result;
    }
}
